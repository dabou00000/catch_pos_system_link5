🚨 حل إجباري لمشكلة تحديث السعر

📋 الخطوات الإجبارية:

  1. افتح Developer Tools (F12) → Console
  2. اذهب إلى نقطة البيع
  3. انسخ والصق هذا الكود في Console:
// حل إجباري لتحديث السعر في نقطة البيع function forceUpdatePOS() { console.log('🔄 بدء التحديث الإجباري...'); // 1. إعادة تحميل المنتجات من localStorage products = JSON.parse(localStorage.getItem('products') || '[]'); console.log('📦 تم تحميل', products.length, 'منتج'); // 2. تحديث كل منتج في العربة if (cart && cart.length > 0) { cart.forEach((cartItem, index) => { const updatedProduct = products.find(p => p.id === cartItem.id); if (updatedProduct) { console.log(`🔄 تحديث منتج ${index + 1}: ${cartItem.name}`); console.log(` السعر القديم: $${cartItem.priceUSD}`); console.log(` السعر الجديد: $${updatedProduct.priceUSD}`); // تحديث جميع البيانات cartItem.name = updatedProduct.name; cartItem.priceUSD = updatedProduct.priceUSD; cartItem.priceLBP = updatedProduct.priceLBP; cartItem.prices = updatedProduct.prices; cartItem.stock = updatedProduct.stock; cartItem.category = updatedProduct.category; cartItem.barcode = updatedProduct.barcode; } }); } // 3. إعادة عرض العربة if (typeof updateCart === 'function') { updateCart(); console.log('✅ تم تحديث العربة'); } // 4. إعادة عرض العربة الأفقي if (typeof updateCartHorizontalDisplay === 'function') { updateCartHorizontalDisplay(); console.log('✅ تم تحديث العرض الأفقي'); } console.log('🎉 تم التحديث الإجباري بنجاح!'); } // تشغيل التحديث forceUpdatePOS();

🔧 حل بديل - تحديث يدوي:

إذا لم يعمل الحل أعلاه، جرب هذا:

// حل يدوي لتحديث سعر منتج معين function updateSpecificProduct(productId, newPriceUSD) { // تحديث في localStorage let products = JSON.parse(localStorage.getItem('products') || '[]'); let product = products.find(p => p.id === productId); if (product) { product.priceUSD = newPriceUSD; product.priceLBP = Math.round(newPriceUSD * 90000); // افتراض سعر صرف 90000 localStorage.setItem('products', JSON.stringify(products)); console.log(`✅ تم تحديث سعر المنتج ${product.name} إلى $${newPriceUSD}`); } // تحديث في العربة if (cart && cart.length > 0) { let cartItem = cart.find(item => item.id === productId); if (cartItem) { cartItem.priceUSD = newPriceUSD; cartItem.priceLBP = Math.round(newPriceUSD * 90000); console.log(`✅ تم تحديث سعر المنتج في العربة`); } } // إعادة عرض العربة if (typeof updateCart === 'function') { updateCart(); } } // مثال: تحديث منتج بمعرف 1 إلى سعر $35 // updateSpecificProduct(1, 35);

⚠️ تحذير مهم:

هذا حل مؤقت. بعد استخدامه، يجب أن نجد السبب الجذري للمشكلة ونصلحه في الكود الأساسي.

🔍 تشخيص المشكلة:

انسخ والصق هذا الكود في Console لتشخيص المشكلة:

// تشخيص مشكلة تحديث السعر function diagnosePriceUpdate() { console.log('🔍 بدء التشخيص...'); // 1. فحص localStorage let storedProducts = JSON.parse(localStorage.getItem('products') || '[]'); console.log('📦 عدد المنتجات في localStorage:', storedProducts.length); // 2. فحص متغير products console.log('📦 عدد المنتجات في المتغير:', products.length); // 3. فحص العربة console.log('🛒 عدد المنتجات في العربة:', cart.length); // 4. فحص منتج معين (غير محدد) let testProduct = storedProducts.find(p => p.name.includes('t-shirt')); if (testProduct) { console.log('🧪 منتج الاختبار:', testProduct.name); console.log(' السعر في localStorage:', testProduct.priceUSD); console.log(' الأسعار المتعددة:', testProduct.prices); } // 5. فحص المنتج في العربة if (cart.length > 0) { let cartProduct = cart.find(item => item.name.includes('t-shirt')); if (cartProduct) { console.log('🛒 المنتج في العربة:', cartProduct.name); console.log(' السعر في العربة:', cartProduct.priceUSD); console.log(' الأسعار المتعددة:', cartProduct.prices); } } console.log('✅ انتهى التشخيص'); } diagnosePriceUpdate();
